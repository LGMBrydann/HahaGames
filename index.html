// Stars animation const canvas = document.getElementById("stars"); const ctx = canvas.getContext("2d"); let w, h; const stars = []; const COUNT = 180; function resize() { w = canvas.width = window.innerWidth; h = canvas.height = window.innerHeight; } window.addEventListener("resize", resize); resize(); for (let i = 0; i < COUNT; i++) { stars.push({ x: Math.random() * w, y: Math.random() * h, r: Math.random() * 1.5 + 0.5, s: Math.random() * 0.6 + 0.2 }); } function animate() { ctx.clearRect(0, 0, w, h); ctx.fillStyle = "white"; for (let s of stars) { ctx.beginPath(); ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2); ctx.fill(); s.y += s.s; if (s.y > h) { s.y = 0; s.x = Math.random() * w; } } requestAnimationFrame(animate); } animate();
